<h3>Usage</h3>
<br />
<h5>Adding DisprzLocalizer Provider</h5>

```jsx
import { DisprzLocalizer } from "@disprz/components";
import EN from "./string/en.json";

function App() {
  const lang = "en";
  return (
    <DisprzLocalizer
      key={lang} // Use the key prop to rerender when language changes
      lang={lang} // You can get this from Skilltron
      defaultTranslation={EN} // You can pass the english json local file here
      mfe={"disprz-components"} // You can get this from package.json
      clientIdentifier={"AmazonEU"}
      terminologyInfoString={{
        en: [
          {
            itemIdentifier: "term_skill_short",
            itemName: "Skill",
            singularForm: "Skill",
            pluralForm: "Skills",
            examples: ["Courses"],
          },
        ],
      }} // You can get this from Skilltron
    >
      {/* Your App goes here  */}
    </DisprzLocalizer>
  );
}
```

<h5>Using `useLocalizerContext`</h5>

```jsx
import { useLocalizerContext } from "@disprz/components";

function MyComponent() {
  const { getLanguageText } = useLocalizerContext();
  return (
    // ....
    <span>{getLanguageText("hello.world")}</span>
    // ....
  );
}
```

<subscript>
  If you were using LocalizerContext from `@disprz/components`, we recommended
  you to migrate to `useLocalizerContext`
</subscript>
